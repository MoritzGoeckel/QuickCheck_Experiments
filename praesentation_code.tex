DEFINITION BUGGY SUM

\begin{minted}{haskell}
buggySum :: [Int] -> Int
buggySum xs = 
    if (elem 7 xs) && (elem 30 xs)
        then (sum xs) - 1
        else sum xs
\end{minted}

MANUELLER TEST

\begin{minted}{haskell}
*Main> buggySum [10, 2]
12
*Main> buggySum [7, 30, 10]
46
\end{minted}

AUTOMATISIERTER TEST

\begin{minted}{haskell}
testSum :: Bool
testSum = 
    sum [1,2] == 3

*Main> testSum
True
\end{minted}

PROPERTY

\begin{minted}{haskell}
propSum :: [Int] -> Bool
propSum xs = 
    sum xs == buggySum xs
\end{minted}
